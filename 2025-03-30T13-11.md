def get_leap_days_count(  
    period_start: datetime.date,  
    period_end: datetime.date,  
) -> int:  
    """  
    Получает количество дней периода (обе границы берутся включительно!)    принадлежащих високосным годам.  
    Например, для промежутка [2023-12-01, 2024-01-31] вернёт 31,    а для промежутка [2020-12-01, 2024-01-31] вернёт 62    """    current_date = period_start  
    leap_days_count = 0  
    while True:  
        year_end = datetime.date(  
            year=current_date.year,  
            month=12,  
            day=31,  
        )  
  
        if calendar.isleap(current_date.year):  
            # количество дней периода в этом году  
            # прибавляем 1, т.к. включаем обе границы            leap_days_count += 1 + (min(year_end, period_end) - current_date).days  
  
        if year_end >= period_end:  
            break  
  
        current_date = datetime.date(  
            year=current_date.year + 1,  
            month=1,  
            day=1,  
        )  
  
    return leap_days_count
